# postinst

## 文件位置

```
# 安装后执行的脚本
/var/lib/dpkg/info/libnginx-mod-http-dav-ext.postinst
```

## 文件内容

```sh
#!/bin/sh
set -e
# dh_nginx/UNDECLARED自动添加的部分
for confpair in mod-http-dav-ext.conf:50-mod-http-dav-ext.conf  ; do
	from=$(echo $confpair | cut -d: -f1)
	to=$(echo $confpair | cut -d: -f2)

    # 删除：/etc/nginx/modules-enabled/50-mod-http-dav-ext.conf
	if [ -L /etc/nginx/modules-enabled/$to.removed ]; then
		rm /etc/nginx/modules-enabled/$to.removed
		removed_link=true
	else
		removed_link=false
	fi

	# 添加符号连接：/etc/nginx/modules-enabled/50-mod-http-dav-ext.conf
	# 1、全新安装
	# 2、自动删除后重新安装（保留管理员操作）
	if [ -z "$2" -o "$removed_link" = "true" ]; then
		ln -sf /usr/share/nginx/modules-available/$from \
			/etc/nginx/modules-enabled/$to
	fi
done

if [ "$1" = "configure" ] ; then
	if which dpkg-trigger >/dev/null 2>&1 ; then
		dpkg-trigger --no-await nginx-reload
	fi

fi
# 自动添加的部分结束
```
